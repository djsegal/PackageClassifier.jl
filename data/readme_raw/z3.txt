Z3.jl This package provides an interface to the Z3 Theorem Prover by wrapping the C++ API of Z3 using CxxWrap.jl . There are still a few parts of the API missing. Feel free to contribute! ctx = Context () x = real_const (ctx, " x " ) y = real_const (ctx, " y " ) s = Solver (ctx, " QF_NRA " ) add (s, x == y ^ 2 ) add (s, x > 1 ) res = check (s) @assert res == Z3 . sat m = get_model (s) for (k,v) in consts (m) println ( " $k = $v " ) end
